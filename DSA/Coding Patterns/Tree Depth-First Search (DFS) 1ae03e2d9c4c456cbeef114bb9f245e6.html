   void traverseInOrder(TreeNode<Integer> root) {
        if (root == null) {
            return;
        }
        var stack = new Stack<TreeNode<Integer>>();
        var currentNode = root;
        while (currentNode != null || !stack.empty()) {
            // Reach the left most TreeNode
            while (currentNode != null) {
                stack.push(currentNode);
                currentNode = currentNode.left;
            }
            // Current must be null at this point
            currentNode = stack.pop();

            visit(currentNode);

            // Move to the right subtree
            currentNode = currentNode.right;
        }
    }

   public void traversePostOrder(TreeNode<Integer> root) {
        if (root == null) {
            return;
        }
        var traverseStack = new Stack<TreeNode<Integer>>();
        var visitingStack = new Stack<TreeNode<Integer>>();

        traverseStack.push(root);

        while (!traverseStack.isEmpty()) {
            // Pop an item from traverseStack and push it to stackForVisitingTheElements
            var node = traverseStack.pop();
            visitingStack.push(node);

            // Push left and right children of removed item to traverseStack
            if (node.left != null) traverseStack.push(node.left);
            if (node.right != null) traverseStack.push(node.right);
        }

        while (!visitingStack.isEmpty()) {
            visit(visitingStack.pop());
        }
    }

Uniqueness

  1. Non-uniqueness of Single Traversals:
    • A single traversal (in-order, pre-order, or post-order) of a BST is not unique. This means that different trees can produce the same traversal result.
    • Example: Consider two degenerate trees (which are essentially linked lists):
      • A right-slanting tree (each node only has a right child).
      • A left-slanting tree (each node only has a left child).
      • Both these trees, despite being structurally different, will yield the same in-order, pre-order, and post-order traversals if they contain the same set of elements.
  1. Uniqueness with Two Traversals:
    • While a single traversal is not enough to uniquely identify a BST, two different types of traversals together can uniquely represent and reconstruct the tree.
    • Common pairs of traversals that can be used to uniquely determine a BST are:
      • In-order and Pre-order: The in-order traversal provides the sorted order of elements, and the pre-order traversal provides the root and the structure of the tree.
      • In-order and Post-order: The in-order traversal gives the sorted order, while the post-order traversal gives the root and the structure in a different manner.
      • Pre-order and Post-order: This combination is less common but can still be used under specific conditions.
