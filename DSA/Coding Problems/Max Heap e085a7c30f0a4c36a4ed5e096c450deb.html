Educative Max Heap Algorithm 

public class Heap {

    private void maxHeapify(int[] heapArray, int indexOfCurrentNode, int heapSize) {

        int indexOfLargestNodeValue = indexOfCurrentNode;

        // check parent nodes only
        // (all parents nodes are placed on left side of the array)
        while (indexOfLargestNodeValue < heapSize / 2) {
            //left child
            int leftChildIndex = (2 * indexOfCurrentNode) + 1;
            //right child
            int rightChildIndex = (2 * indexOfCurrentNode) + 2;

            // leftChildIndex < heapSize : check if leftChildIndex within the bounds of the heap.
            //      If the left child index is greater than or equal to heapSize,
            //      it means the node does not have a left child, and we should not access heapArray[leftChildIndex]
            // check if the value of leftChild node is Greater than current node value
            if (leftChildIndex < heapSize
                    && heapArray[leftChildIndex] > heapArray[indexOfCurrentNode]) {
                // update the largest index to current left child index
                indexOfLargestNodeValue = leftChildIndex;
            }

            /*
                rightChildIndex < heapSize:
                            this checks if the right child index is within the bounds of the heap.
                            If the right child index is greater than or equal to heapSize,
                            it means the node does not have a right child, and we should not
                             access heapArray[rightChildIndex]
             */
            if (rightChildIndex < heapSize && heapArray[rightChildIndex] > heapArray[indexOfLargestNodeValue]) {
                // update the largest index to current RIGHT child index
                indexOfLargestNodeValue = rightChildIndex;
            }
            if (indexOfLargestNodeValue != indexOfCurrentNode) { // swap parent with largest child
                int temp = heapArray[indexOfCurrentNode];
                heapArray[indexOfCurrentNode] = heapArray[indexOfLargestNodeValue];
                heapArray[indexOfLargestNodeValue] = temp;
                indexOfCurrentNode = indexOfLargestNodeValue;
            } else
                break; // if heap property is satisfied
        } //end of while
    }

    public void buildMaxHeap(int[] heapArray, int heapSize) {
        // swap largest child to parent node
        // starts from the mid of the array
        int mid = (heapSize - 1) / 2;
        
        // visit each node at index starting from mid to 0
        for (int i = mid; i >= 0; i--) {
            maxHeapify(heapArray, i, heapSize);
        }
    }

}
