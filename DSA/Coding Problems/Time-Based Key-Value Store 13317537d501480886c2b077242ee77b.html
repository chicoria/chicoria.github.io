class TimeStamp {

    Map<String, List<String>> valuesMap = new HashMap<>();
    Map<String, List<Integer>> timestampsMap = new HashMap<>();
    public TimeStamp() {
        // Write your code here
    }

    // Set TimeStamp data variables
    public boolean setValue(String key, String value, int timestamp) {
        if (valuesMap.containsKey(key)) {
            var existentValue = valuesMap.get(key);
            var lastValueIndex = existentValue.size() - 1;
            if (!value.equals(existentValue.get(lastValueIndex))) {
                existentValue.add(value);
                timestampsMap.get(key).add(timestamp);
            }
        } else {
            valuesMap.put(key, new ArrayList<>());
            valuesMap.get(key).add(value);
            timestampsMap.put(key, new ArrayList<>());
            timestampsMap.get(key).add(timestamp);
        }
        return true;
    }

    // Get the value for the given key and timestamp
    public String getValue(String key, int timeStamp) {
        int index = 0;
        if (!valuesMap.containsKey(key)) {
            return "";
        } else {
            index = searchRightMostTimestamp(timestampsMap.get(key), timeStamp) ;
        }

        if (index > -1) {
            return valuesMap.get(key).get(index);
        }
        return "";
    }

    private int searchRightMostTimestamp(List<Integer> list, int timestamp){
        int left = 0;
        int right = list.size();
        int mid = 0;
        while ( left < right){
            mid =  left + (right - left) / 2;;

            if (!(list.get(mid) > timestamp)){
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left - 1;
    }
}
