class SnapshotArray {
    int snapshotId;
    int length;
    private Map<Integer, Map<Integer, Integer>> snapshots;

    // Constructor
    public SnapshotArray(int length) {
        this.snapshotId = 0;
        this.snapshots = new HashMap<>();
        this.snapshots.put(0, new HashMap<>(length));
        this.length = length;
    }

    // Function setValue sets the value at a given index idx to val.
    public void setValue(int idx, int state) {
        if (idx < length){
            snapshots.get(snapshotId).put(idx, state);
        }
    }

    // This function takes no parameters and returns the snapid.
    // snapid is the number of times that the snapshot() function was called minus 1.
    public int snapshot() {
        this.snapshots.put(snapshotId + 1, new HashMap<>(this.snapshots.get(snapshotId)));
        snapshotId++;
        return snapshotId - 1;
    }

    // Function getValue returns the value at the index idx with the given snapid.
    public int getValue(int idx, int snapshotId1) {
        if(snapshotId1 < length || snapshotId1 >= 0){
            return snapshots.get(snapshotId1).getOrDefault(idx, 0);
        } else {
            return -1;
        }
    }
}
