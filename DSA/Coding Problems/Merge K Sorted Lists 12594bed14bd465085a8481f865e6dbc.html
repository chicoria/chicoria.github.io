    public static LinkedListNode<Integer> mergeTwoListsDivideAndConquer(LinkedListNode<Integer> head1, LinkedListNode<Integer> head2) {
        LinkedListNode<Integer> dummy = new LinkedListNode<Integer>(-1);
        LinkedListNode<Integer> prev = dummy;

        while (head1 != null && head2 != null) {
            if (head1.data<= head2.data) {
                prev.next = head1;
                head1 = head1.next;
            } else {
                prev.next = head2;
                head2 = head2.next;
            }
            prev = prev.next;
        }

        if (head1 == null)
            prev.next = head2;
        else
            prev.next = head1;

        return dummy.next;
    }


    public static LinkedListNode<Integer> mergeKListsGpt(List<LinkedList> lists) {
        if (lists == null || lists.isEmpty()) return null;
        return mergeKListsHelper(lists, 0, lists.size() - 1);
    }

    private static LinkedListNode<Integer> mergeKListsHelper(List<LinkedList> lists, int start, int end) {
        if (start == end) {
            System.out.println("Returning list from index: " + start);
            printList(lists.get(start).head);
            return lists.get(start).head;
        }

        int mid = start + (end - start) / 2;
        System.out.println("Dividing lists from index " + start + " to " + end);
        LinkedListNode<Integer> left = mergeKListsHelper(lists, start, mid);
        LinkedListNode<Integer> right = mergeKListsHelper(lists, mid + 1, end);

        System.out.println("Merging lists:");
        printList(left);
        printList(right);

        LinkedListNode<Integer> merged = mergeTwoListsDivideAndConquer(left, right);
        System.out.println("Merged result:");
        printList(merged);
        return merged;
    }
