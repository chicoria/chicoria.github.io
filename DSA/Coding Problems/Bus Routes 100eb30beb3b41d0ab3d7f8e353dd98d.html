    public static int minimumBuses(int[][] busRoutes, int src, int dest) {
        if (src == dest){
            return 0;
        }
        var busByStation = new HashMap<Integer, List<Integer>>(); //station, bus
        for (int i=0; i < busRoutes.length; i++){
            for (int j = 0; j < busRoutes[i].length; j++) {
                busByStation.computeIfAbsent(busRoutes[i][j], key -> new ArrayList<>() ).add(i);
            }
        }
        var busVisited = new HashSet<Integer>();

        Queue<int[]> q = new ArrayDeque<>();
        q.offer(new int[]{src, 0});

        while (!q.isEmpty()){
            var currentStationAndCountBus = q.poll();
            var currentStation = currentStationAndCountBus[0];
            var busTaken = currentStationAndCountBus[1];
            if (currentStation == dest ){
                return busTaken;
            }
            var busList = busByStation.getOrDefault(currentStation, new ArrayList<>());
            for (var bus : busList) {
                if (busVisited.contains(bus)) continue;;
                busVisited.add(bus);

                for (var station:  busRoutes[bus]) {
                    q.offer(new int[]{station, busTaken +1});
                }
            }
        }
        return -1;
    }
