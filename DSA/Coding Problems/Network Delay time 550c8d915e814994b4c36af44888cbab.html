public static int networkDelayTime(int[][] times, int n, int k) {
        Map<Integer, List<int[]>> adjacency = new HashMap<>();
        for (int[] time : times) {
            int source = time[0];
            int destination = time[1];
            int travelTime = time[2];
            adjacency.computeIfAbsent(source, key -> new ArrayList<>()).add(new int[]{destination, travelTime});
        }

        PriorityQueue<int[]> pq = new PriorityQueue<>(Comparator.comparingInt(a -> a[0]));
        pq.offer(new int[]{0, k});
        Set<Integer> visited = new HashSet<>();
        int delays = 0;

        while (!pq.isEmpty()) {
            int[] current = pq.poll();
            int time = current[0];
            int node = current[1];

            if (visited.contains(node))
                continue;

            visited.add(node);
            delays = Math.max(delays, time);
            List<int[]> neighbors = adjacency.getOrDefault(node, new ArrayList<>());

            for (int[] neighbor : neighbors) {
                int neighborNode = neighbor[0];
                int neighborTime = neighbor[1];
                if (!visited.contains(neighborNode)) {
                    int newTime = time + neighborTime;
                    pq.offer(new int[]{newTime, neighborNode});
                }
            }
        }

        if (visited.size() == n)
            return delays;

        return -1;
    }

    public static int networkDelayTime(int[][] times, int n, int k) {
        int delays = 0;
        var map = new HashMap<Integer, List<NetworkNode>>();
        for (var time : times){
            int source = time[0];
            int destination = time[1];
            int travelTime = time[2];
            map.computeIfAbsent(source, key -> new ArrayList<>()).add(new NetworkNode(destination, travelTime));
        }
        Queue<NetworkNode> nodesWithMinimalDelayTime = new PriorityQueue<NetworkNode>( (node1, node2) -> node1.delayTime - node2.delayTime);
        var visitedNodesSet = new HashSet<Integer>();
        
        nodesWithMinimalDelayTime.offer(new NetworkNode(k,0));

        while(!nodesWithMinimalDelayTime.isEmpty()){
            var currentNode = nodesWithMinimalDelayTime.poll();
            if (visitedNodesSet.contains(currentNode.value)){
                continue;
            }
            visitedNodesSet.add(currentNode.value);
            delays = Math.max(delays,currentNode.delayTime);

            var neighbors = map.getOrDefault(currentNode.value, new ArrayList<>());

            for (var neighborNode : neighbors){
                if (!visitedNodesSet.contains(neighborNode.value)){
                    neighborNode.delayTime = currentNode.delayTime + neighborNode.delayTime;;
                    nodesWithMinimalDelayTime.offer(neighborNode);
                }
            }
        }
        
        if (visitedNodesSet.size() == n){
            return delays;
        }
        
        return -1;
    }
    
class NetworkNode {
    Integer value;
    Integer delayTime;

    public NetworkNode(int node, int delay){
        this.delayTime = delay;
        this.value = node;
    }
